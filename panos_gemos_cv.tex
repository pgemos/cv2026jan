% Command for drawing a 5-star skill bar (Left-to-Right)
% Usage: \starskill{Skill Name}{3}
\newcommand{\starskill}[2]{%
    \noindent
    \begin{tikzpicture}
        % Skill Name
        \node [anchor=west] at (0,0) {\small #1};
        
        % Draw 5 Stars
        \foreach \x in {1,...,5} {
            % Calculate position (starting from 2.5cm to the right of text)
            \pgfmathsetmacro\xpos{2.3 + (\x * 0.4)}
            
            \ifnum \x > #2
                % Empty star (outline) for x > rating
                \node [anchor=center] at (\xpos, 0) {\small \color{gray}\faStar[regular]};
            \else
                % Filled star for x <= rating
                \node [anchor=center] at (\xpos, 0) {\small \color{accentcolor}\faStar};
            \fi
        }
    \end{tikzpicture}\\[2pt]
}
